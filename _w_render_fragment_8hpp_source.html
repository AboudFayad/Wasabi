34  void OnEntityChange(EntityT* entity, bool added) {
35  if (added) {
36  OnEntityAdded(entity);
37  SortingKeyT key(entity);
38  m_allEntities.insert(std::make_pair(key, entity));
39  } else {
40  auto iter = m_allEntities.find(SortingKeyT(entity));

43  else {
44  // the entity seems to have changed and then removed before we cloud reindex it in the render loop
45  // need to find it manually now...

47  if (it->second == entity) {

97  virtual WError Render(class WRenderer* renderer, class WRenderTarget* rt) {
98  UNREFERENCED_PARAMETER(renderer);
99 
100  WEffect* boundFX = nullptr;
101  for (auto it = m_allEntities.begin(); it != m_allEntities.end(); it++) {
102  EntityT* entity = it->second;
103  if (ShouldRenderEntity(entity)) {
104  WEffect* effect = m_renderEffect;
105  WMaterial* material = entity->GetMaterial(effect);
106  if (!material) {
107  // see if a custom effect can be used
108  for (auto mat : entity->GetMaterials().m_materials) {
109  if (mat.first->GetEffect()->GetRenderFlags() & m_requiredRenderFlags) {
110  material = mat.first;
111  effect = material->GetEffect();
112  break;
113  }
114  }
115  }
116  if (material && entity->WillRender(rt)) {
117  if (boundFX != effect) {
118  effect->Bind(rt);
119  boundFX = effect;
120  }
121  if (KeyChanged(entity, effect, it->first))
122  m_reindexEntities.push_back(std::make_pair(it->first, effect));
123 
124  RenderEntity(entity, rt, material);
125  }
126  }
127  }
128  for (auto it : m_reindexEntities) {
129  m_allEntities.erase(it.first);
130  m_allEntities.insert(std::make_pair(SortingKeyT(it.first.GetEntity(), it.second), it.first.GetEntity()));
131  }
132  m_reindexEntities.clear();
133 
134  return WError(W_SUCCEEDED);
135  }
